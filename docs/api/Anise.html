Anise

Dynamic Annotations for Ruby.

require 'anise'

class X
  include Anise

  # Provides annotations
  ann :foo, :class=>String

  # Provides annotated attributes
  attr :bar, Integer, :max=>10

  # Provides method annotators.
  annotator :doc
  doc "foo is cool"
  def foo
    # ...
  end
end
